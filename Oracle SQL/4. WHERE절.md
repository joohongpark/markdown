# WHERE 절

## 테이블 정의

테이블 내에 원하는 데이터만 제어할 때 WHERE 절을 잘 활용해야 한다.

WHERE 절에는 대소, 등가, AND, OR,  NOT 연산 등을 지원한다.

```plsql
SELECT * FROM EMPLOYEES WHERE LAST_NAME = 'Fay'; -- equal
SELECT * FROM EMPLOYEES WHERE LAST_NAME <> 'Fay'; -- not equal
SELECT * FROM EMPLOYEES WHERE SALARY > 10000; -- 대소
SELECT * FROM EMPLOYEES WHERE SALARY <= 10000; -- 대소1
SELECT * FROM EMPLOYEES WHERE SALARY BETWEEN 1000 AND 10000; -- 범위 값
SELECT * FROM EMPLOYEES WHERE SALARY IN (9000, 6000); -- 특정 값에 속하는 컬럼
SELECT * FROM EMPLOYEES WHERE LAST_NAME LIKE 'H%'; -- 특정 값에 속하는 컬럼
SELECT * FROM EMPLOYEES WHERE SALARY < 20000 AND SALARY > 10000; -- AND 연산
SELECT * FROM EMPLOYEES WHERE SALARY > 20000 OR SALARY < 10000; -- AND 연산
SELECT * FROM EMPLOYEES WHERE NOT SALARY > 10000; -- NOT 연산
SELECT * FROM EMPLOYEES WHERE NOT (SALARY > 20000 OR SALARY < 10000); -- NOT 연산 1
SELECT * FROM EMPLOYEES WHERE
        DEPARTMENT_ID IN (
            SELECT DEPARTMENT_ID FROM DEPARTMENTS WHERE DEPARTMENT_NAME = 'IT'
            ); -- SELECT 문 중첩, IN 활용
```
